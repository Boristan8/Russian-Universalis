namespace = empirefactions

# Food Shortage, choose between ADM and DIP
country_event = {
	id = empirefactions.1
	title = "EmpFact1NAME"
	desc = "EmpFact1DESC"
	picture = MERCHANTS_TALKING_eventPicture
	
	trigger = {
		OR = {
			government = m_empire
			government = d_empire
			government = r_empire
		}
	}
	
	mean_time_to_happen = {
		months = 800
	}
	
	option = {
		name = "EmpFact1OPT1"
		add_faction_influence = {
			faction = emp_adm
			influence = 20
		}
		add_faction_influence = {
			faction = emp_dip
			influence = -20
		}
	}
	
	option = {
		name = "EmpFact1OPT2"
		add_faction_influence = {
			faction = emp_dip
			influence = 20
		}
		add_faction_influence = {
			faction = emp_adm
			influence= -20
		}
	}
}

#War Exhaustion, trade MIL for ADM
country_event = {
	id = empirefactions.2
	title = "EmpFact2NAME"
	desc = "EmpFact2DESC"
	picture = ECONOMY_eventPicture
	
	trigger = {
		OR = {
			government = m_empire
			government = d_empire
			government = r_empire
		}
		war_exhaustion = 5
	}
	
	mean_time_to_happen = {
		months = 48
	}
	
	option = {
		name = "EmpFact2OPT1"
		add_faction_influence = {
			faction = emp_adm
			influence = 20
		}
		add_faction_influence = {
			faction = emp_mil
			influence = -20
		}
		add_war_exhaustion = -5
	}

	option = {
		name = "EmpFact2OPT2"
		add_war_exhaustion = 5
	}
}
