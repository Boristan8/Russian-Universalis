# From Novgorod mod by Skepteroun
subjugate_viatka = {
	type = country
	category = MIL
	
	target_provinces = {
		owned_by = KHY
	}
	allow = {
		normal_or_historical_nations = yes
		OR = {
			tag = MOS
			tag = TSR
			tag = RUS
		}
		is_subject = no
		KHY = {
			is_neighbor_of = ROOT
			NOT = { num_of_cities = ROOT }
			religion_group = ROOT
			NOT = { is_subject_of = ROOT }
		}
		NOT = { has_country_flag = viatka_subjugated }
		NOT = { last_mission = subjugate_viatka }
	}
	abort = {
		OR = {
			is_subject = yes
			NOT = { exists = KHY }
			NOT = { religion_group = ROOT }
		}
	}
	success = {
		OR = {
			KHY = { is_subject_of = ROOT }
			AND = {
				NOT = { exists = KHY }
				1079 = { owned_by = ROOT }
			}
		}
	}
	chance = {
		factor = 5000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = KHY value = 0 } }
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			remove_claim = ROOT
		}
	}
	effect = {
		add_prestige = 5
		if = {
			limit = {
				any_target_province = {
					province_id = 1079
				}
			}
			1079 = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
		if = {
			limit = {
				NOT = {
					any_target_province = {
						province_id = 1079
					}
				}
			}
			random_target_province = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1				
			}
		}
		set_country_flag = viatka_subjugated
	}
}

subjugate_perm = {

	type = country
	
	category = DIP
	ai_mission = yes
	
	target_provinces = {
		owned_by = PRM
	}
	allow = {
		normal_or_historical_nations = yes
		num_of_cities = 3
		primary_culture = russian
		PRM = {
			ai = yes
			exists = yes
			is_subject = no
			num_of_cities = 2
			is_neighbor_of = ROOT
		}
		is_subject = no
		NOT = { war_with = PRM }
		NOT = { has_country_flag = perm_mission_rus }
		NOT = { last_mission = subjugate_perm }
	}
	abort = {
		OR = {
			is_subject = yes
			PRM = {
				is_subject = yes
				NOT = { is_subject_of = ROOT }
			}
		}
	}
	success = {
		is_at_war = no
		OR = {
			PRM = { is_subject_of = ROOT }
			PRM = {
				has_opinion = {
					who = ROOT
					value = 175
				}
			}
		}
		PRM = {
			is_at_war = no
			is_neighbor_of = ROOT
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			dip = 3
		}	
	}
	effect = {
		set_country_flag = perm_mission_rus
		add_prestige = 5
		vassalize = PRM
	}
}
