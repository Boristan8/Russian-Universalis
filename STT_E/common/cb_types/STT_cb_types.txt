cb_panslavism = {
	
	prerequisites = {
		OR = {
			adm_tech = 20
			tag = SLV
		}
		OR = {
			culture_group = west_slavic
			culture_group = east_slavic
		}
		FROM = {
			NOT = { culture_group = west_slavic }
			NOT = { culture_group = east_slavic }
			NOT = { culture_group = south_slavic }
			any_owned_province = {
				OR = {
					culture_group = west_slavic
					culture_group = east_slavic
					culture_group = south_slavic
				}
			}
		}
	}
	
	war_goal = take_slavic_province
}
